    Content loading per D-05 and D-06:
    ```javascript
    let contentCache = [];

    async function loadContentDiscovery() {
      const grid = document.getElementById('content-grid');
      grid.innerHTML = '<div class="loading-state">Loading content...</div>';

      const sourceFilter = document.getElementById('source-filter').value;
      const statusFilter = document.getElementById('status-filter').value;

      try {
        let items = [];
        if (statusFilter) {
          // Single status selected
          let url = `/api/content?status=${statusFilter}&limit=200`;
          if (sourceFilter) url += `&source=${sourceFilter}`;
          const resp = await authFetch(url);
          if (resp.ok) items = await resp.json();
        } else {
          // Default: both discovered + surfaced per D-05
          const [r1, r2] = await Promise.all([
            authFetch(`/api/content?status=discovered&limit=200${sourceFilter ? '&source=' + sourceFilter : ''}`),
            authFetch(`/api/content?status=surfaced&limit=200${sourceFilter ? '&source=' + sourceFilter : ''}`)
          ]);
          if (r1.ok) items = items.concat(await r1.json());
          if (r2.ok) items = items.concat(await r2.json());
          items.sort((a, b) => new Date(b.discovered_at) - new Date(a.discovered_at));
        }

        contentCache = items;

        if (items.length === 0) {
          grid.innerHTML = '<div class="empty-state">No content found. Check filters or wait for next scrape run.</div>';
          return;
        }

        grid.innerHTML = items.map(renderContentCard).join('');
      } catch (e) {
        grid.innerHTML = '<div class="empty-state">Failed to load content. Check connection.</div>';
        console.error('Content load error:', e);
      }
    }
    ```

    function safeParseMeta(item) {
      try { return JSON.parse(item.metadata_json || '{}'); } catch { return {}; }
    }

    function renderContentCard(item) {
      const meta = safeParseMeta(item);
      const thumb = item.thumbnail_url
        ? `<img class="content-thumb" src="${item.thumbnail_url}" loading="lazy" onclick="openPreviewModal('${item.id}')">`
        : `<div class="content-thumb content-thumb-placeholder" onclick="openPreviewModal('${item.id}')">No preview</div>`;

      const stats = [];
      if (meta.views) stats.push(`${Number(meta.views).toLocaleString()} views`);
      if (meta.likes) stats.push(`${Number(meta.likes).toLocaleString()} likes`);
      if (meta.engagement_rate) stats.push(`${(meta.engagement_rate * 100).toFixed(1)}% ER`);

      return `
        <div class="content-card" data-id="${item.id}" data-status="${item.status}">
          ${thumb}
          <div class="content-body">
            <div style="display:flex;gap:6px;margin-bottom:8px;flex-wrap:wrap">
              <span class="source-badge source-${item.source}">${item.source}</span>
              <span class="source-badge status-${item.status}">${item.status.replace('_', ' ')}</span>
            </div>
            <div style="font-size:0.85rem;color:var(--text);margin-bottom:4px;overflow:hidden;text-overflow:ellipsis;display:-webkit-box;-webkit-line-clamp:2;-webkit-box-orient:vertical">${item.ad_copy || 'No ad copy'}</div>
            <div class="content-meta">
              ${stats.length ? stats.join(' &middot; ') + ' &middot; ' : ''}${new Date(item.discovered_at).toLocaleDateString()}
            </div>
          </div>
          <div class="content-actions">
            ${renderCardActions(item)}
          </div>
        </div>`;
    }

    Lifecycle action buttons per D-09 and D-10:
    ```javascript
    function renderCardActions(item) {
      const dl = item.creative_url
        ? `<a class="content-btn secondary" href="${item.creative_url}" target="_blank" rel="noopener" onclick="event.stopPropagation()">Download</a>`
        : '';

      if (item.status === 'discovered') {
        return `<button class="content-btn" onclick="event.stopPropagation();advanceStatus('${item.id}','surfaced')">Surface</button>${dl}`;
      }
      if (item.status === 'surfaced') {
        return `<button class="content-btn" onclick="event.stopPropagation();advanceStatus('${item.id}','queued')">Queue for Remake</button>${dl}`;
      }
      if (item.status === 'queued') {
        return `
          <input type="text" class="drive-link-input" id="drive-${item.id}" placeholder="Paste Google Drive link..." onclick="event.stopPropagation()">
          <button class="content-btn" onclick="event.stopPropagation();submitDriveLink('${item.id}')">Submit Remake</button>${dl}`;
      }
      if (item.status === 'ready_to_launch') {
        return `<span style="color:var(--accent);font-size:0.8rem">Ready for launch</span>${dl}`;
      }
      return dl;
    }

    async function advanceStatus(id, newStatus) {
      try {
        const resp = await authFetch(`/api/content/${id}/status`, {
          method: 'PATCH',
          headers: {'Content-Type': 'application/json'},
          body: JSON.stringify({ status: newStatus })
        });
        if (resp.ok) {
          loadContentDiscovery();
        } else {
          const err = await resp.json();
          alert(err.detail || 'Failed to update status');
        }
      } catch (e) {
        console.error('Status update error:', e);
        alert('Failed to update status');
      }
    }

    async function submitDriveLink(id) {
      const input = document.getElementById('drive-' + id);
      const link = input ? input.value.trim() : '';
      if (!link) { alert('Please paste a Google Drive link'); return; }
      if (!link.includes('drive.google.com') && !link.includes('docs.google.com')) {
        if (!confirm('This does not look like a Google Drive link. Submit anyway?')) return;
      }
      try {
        const resp = await authFetch(`/api/content/${id}/status`, {
          method: 'PATCH',
          headers: {'Content-Type': 'application/json'},
          body: JSON.stringify({ status: 'ready_to_launch', drive_link: link })
        });
        if (resp.ok) {
          loadContentDiscovery();
        } else {
          const err = await resp.json();
          alert(err.detail || 'Failed to submit remake');
        }
      } catch (e) {
        console.error('Drive link submit error:', e);
        alert('Failed to submit remake');
      }
    }
    ```

    ```javascript
    function openPreviewModal(itemId) {
      const item = contentCache.find(i => i.id === itemId);
      if (!item) return;
      const meta = safeParseMeta(item);
      const isVideo = item.source === 'ppspy' || meta.type === 'video';
      const slides = meta.slides || meta.carousel_images || [];

      let mediaHtml;
      if (slides.length > 1) {
        mediaHtml = `<div style="display:flex;gap:8px;overflow-x:auto;max-width:80vw">${slides.map(s => `<img style="max-height:70vh;border-radius:8px" src="${s}">`).join('')}</div>`;
      } else if (isVideo && item.creative_url) {
        mediaHtml = `<video controls autoplay style="max-width:80vw;max-height:70vh;border-radius:8px" src="${item.creative_url}"></video>`;
      } else {
        const imgSrc = item.creative_url || item.thumbnail_url;
        mediaHtml = imgSrc
          ? `<img style="max-width:80vw;max-height:70vh;object-fit:contain;border-radius:8px" src="${imgSrc}">`
          : '<div class="empty-state">No preview available</div>';
      }

      const content = document.getElementById('preview-modal-content');
      content.innerHTML = `
        ${mediaHtml}
        <div style="margin-top:12px;display:flex;gap:8px;justify-content:center">
          ${item.creative_url ? `<a class="content-btn" href="${item.creative_url}" target="_blank" rel="noopener">Open Original</a>` : ''}
          <button class="content-btn secondary" onclick="closePreviewModal()">Close</button>
        </div>
        <div style="margin-top:8px;font-size:0.8rem;color:var(--text-dim);text-align:center">
          ${item.source.toUpperCase()} &middot; ${item.status} &middot; ${new Date(item.discovered_at).toLocaleDateString()}
        </div>`;

      document.getElementById('preview-modal').classList.remove('hidden');
    }

    function closePreviewModal() {
      document.getElementById('preview-modal').classList.add('hidden');
      const content = document.getElementById('preview-modal-content');
      content.innerHTML = '';
      // Stop any playing video
      content.querySelectorAll('video').forEach(v => { v.pause(); v.src = ''; });
    }

    // Close modal on Escape key
    document.addEventListener('keydown', e => {
      if (e.key === 'Escape') closePreviewModal();
    });

    // Close modal on overlay click
    document.getElementById('preview-modal')?.addEventListener('click', e => {
      if (e.target.classList.contains('preview-overlay')) closePreviewModal();
    });
    ```
