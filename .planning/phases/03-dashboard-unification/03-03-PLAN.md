    ```javascript
    async function loadHealthPanel() {
      const grid = document.getElementById('health-grid');
      grid.innerHTML = '<div class="loading-state">Loading health data...</div>';

      try {
        const resp = await authFetch('/api/content/health');
        if (!resp.ok) throw new Error('Health endpoint returned ' + resp.status);
        const data = await resp.json();

        const sourceNames = { ppspy: 'PPSpy', tiktok: 'TikTok', pinterest: 'Pinterest', meta: 'Meta Ads' };
        const sourceColors = { ppspy: '#ff6b6b', tiktok: '#69c9d0', pinterest: '#e60023', meta: '#1877f2' };

        grid.innerHTML = Object.entries(data).map(([source, info]) => {
          const status = info.ok ? 'ok' : (info.last_seen ? 'stale' : 'down');
          const statusLabel = info.ok ? 'Active today' : (info.last_seen ? 'Stale' : 'No data');
          const lastSeen = info.last_seen
            ? new Date(info.last_seen).toLocaleString()
            : 'Never';
          const timeSince = info.last_seen
            ? timeAgo(new Date(info.last_seen))
            : '—';

          return `
            <div class="health-card ${status}">
              <div class="health-indicator ${status}"></div>
              <div class="health-source-name" style="color:${sourceColors[source] || 'var(--text)'}">${sourceNames[source] || source}</div>
              <div class="health-stat">
                <span class="health-stat-label">Status</span>
                <span class="health-stat-value">${statusLabel}</span>
              </div>
              <div class="health-stat">
                <span class="health-stat-label">Items today</span>
                <span class="health-stat-value">${info.today_count}</span>
              </div>
              <div class="health-stat">
                <span class="health-stat-label">Last discovery</span>
                <span class="health-stat-value">${lastSeen}</span>
              </div>
              <div class="health-stat">
                <span class="health-stat-label">Time since</span>
                <span class="health-stat-value">${timeSince}</span>
              </div>
            </div>`;
        }).join('');

        document.getElementById('health-refresh-time').textContent =
          'Last refreshed: ' + new Date().toLocaleTimeString();

      } catch (e) {
        grid.innerHTML = '<div class="empty-state">Failed to load health data.</div>';
        console.error('Health panel error:', e);
      }
    }

    function timeAgo(date) {
      const seconds = Math.floor((new Date() - date) / 1000);
      if (seconds < 60) return 'just now';
      const minutes = Math.floor(seconds / 60);
      if (minutes < 60) return minutes + 'm ago';
      const hours = Math.floor(minutes / 60);
      if (hours < 24) return hours + 'h ago';
      const days = Math.floor(hours / 24);
      return days + 'd ago';
    }
    ```
