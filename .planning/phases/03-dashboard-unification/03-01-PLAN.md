    CRITICAL: This route MUST be defined BEFORE any potential `GET /api/content/{item_id}` route to avoid FastAPI path collision where "health" is interpreted as an item_id. Currently no such route exists, but place it right after the `list_content_items` function and before the `update_status` function.

    Implementation:
    ```python
    from datetime import datetime, timezone
    from sqlalchemy import func

    @router.get("/api/content/health")
    def content_health(db: Session = Depends(get_db)):
        sources = ["ppspy", "tiktok", "pinterest", "meta"]
        today_start = datetime.now(timezone.utc).replace(hour=0, minute=0, second=0, microsecond=0)
        result = {}
        for source in sources:
            last = (
                db.query(ContentItem)
                .filter_by(source=source)
                .order_by(ContentItem.discovered_at.desc())
                .first()
            )
            today_count = (
                db.query(func.count(ContentItem.id))
                .filter(
                    ContentItem.source == source,
                    ContentItem.discovered_at >= today_start,
                )
                .scalar()
            )
            result[source] = {
                "last_seen": last.discovered_at.isoformat() if last and last.discovered_at else None,
                "today_count": today_count,
                "ok": last is not None and last.discovered_at is not None and last.discovered_at >= today_start if last else False,
            }
        return result
    ```

    NOTE on SQLite vs Postgres datetime: The `discovered_at` column uses `server_default=func.now()`.
